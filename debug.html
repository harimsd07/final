def longest_common_prefix(strings):
            if not strings:
                return ""
            prefix = strings[0]
            for string in strings[1:]:
                while string.find(prefix)!=0:
                    if not prefix:
                        return ""
            return prefix
        print(longest_common_prefix(["flower","flow","flight"]))

DEBUG the following C code:

#include <stdio.h>
            #include 
            
            char* longestCommonPrefix(char** strings, int size) {
                if (size == 0)
                    return "";
            
                char* prefix = strings[0];
                for (int i = 1; i < size; i++) {
                    while (strstr(strings[i], prefix) != strings[i]) {
                        if (!*prefix)
                            return "";
                        prefix[strlen(prefix) - 1] = '\0';
                    }
                }
                return prefix;
            }
            
            int main() {
                char* strings[] = {"flower", "flow", "flight"};
                int size = sizeof(strings) / sizeof(strings[0]);
                printf("%s\n", longestCommonPrefix(strings, size));
                return 0;
            }

DEBUG the following Java code:

  public class Main {
            public static String longestCommonPrefix(String[] strings) {
                if (strings.length == 0)
                    return "";
        
                String prefix = strings[0];
                for (int i = 1; i < strings.length; i++) {
                    while (!strings[i].startsWith(prefix)) {
                        if (prefix.isEmpty())
                            return "";
                        prefix = prefix.substring(0, prefix.length() - 1);
                    }
                }
                return prefix;
            }
        
            public static void main(String[] args) {
                String[] strings = {"flower", "flow", "flight"};
                System.out.println(longestCommonPrefix(strings));
            }
        }
